- Termux 架设手机 Server 下载文件

      # 连接远程仓库，获取软件包信息
      $ apt update

      # 更新本地已经安装的软件包
      $ apt upgrade

      # 安装 sl 软件包
      $ apt install sl

      # 运行
      $ sl

      # 访问本机存储
      $ termux-setup-storage

      # 安装 Node.js
      $ apt install nodejs

      # 安装 npm 模块http-server
      $ npm install -g http-server

      # 运行 Server
      $ http-server
